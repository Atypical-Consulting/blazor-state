Create Subscription.cs with selector-based filtering:
```csharp
namespace Bustand.Core;

/// <summary>
/// Internal subscription implementation with selector-based change detection.
/// </summary>
internal class Subscription<TState, TSlice> : ISubscription where TState : class
{
    private readonly Func<TState, TSlice> _selector;
    private readonly Action _callback;
    private readonly Action<ISubscription> _onDispose;
    private TSlice? _lastValue;
    private bool _isActive = true;
    private bool _isFirstNotification = true;

    public bool IsActive => _isActive;

    public Subscription(
        Func<TState, TSlice> selector,
        Action callback,
        Action<ISubscription> onDispose,
        TState initialState)
    {
        _selector = selector;
        _callback = callback;
        _onDispose = onDispose;
        _lastValue = selector(initialState);
    }

    /// <summary>
    /// Called when store state changes. Only invokes callback if selected slice changed.
    /// </summary>
    public void NotifyIfChanged(TState newState)
    {
        if (!_isActive) return;

        var newValue = _selector(newState);

        // Use reference equality for records (per CONTEXT.md decision)
        if (_isFirstNotification || !ReferenceEquals(_lastValue, newValue))
        {
            _lastValue = newValue;
            _isFirstNotification = false;
            _callback();
        }
    }

    public void Dispose()
    {
        if (_isActive)
        {
            _isActive = false;
            _onDispose(this);
        }
    }
}

/// <summary>
/// Subscription for full state (no selector, always notifies).
/// </summary>
internal class FullStateSubscription<TState> : ISubscription where TState : class
{
    private readonly Action _callback;
    private readonly Action<ISubscription> _onDispose;
    private bool _isActive = true;

    public bool IsActive => _isActive;

    public FullStateSubscription(Action callback, Action<ISubscription> onDispose)
    {
        _callback = callback;
        _onDispose = onDispose;
    }

    public void Notify()
    {
        if (_isActive) _callback();
    }

    public void Dispose()
    {
        if (_isActive)
        {
            _isActive = false;
            _onDispose(this);
        }
    }
}
```

Key design decisions:
- Reference equality per CONTEXT.md (works with C# records)
- Separate FullStateSubscription for Subscribe() without selector (always notifies)
- _onDispose callback lets store remove subscription from tracking list

   ```csharp
   protected virtual void OnStateChanged()
   {
       StateChanged?.Invoke(this, EventArgs.Empty);

       List<ISubscription> subs;
       lock (_subscriptionLock)
       {
           subs = _subscriptions.ToList();
       }

       var state = State;
       foreach (var sub in subs)
       {
           try
           {
               if (sub is IInternalSubscription<TState> internalSub)
               {
                   internalSub.NotifyStateChanged(state);

           catch (ObjectDisposedException)
           {
               // Component disposed during notification - silently ignore per CONTEXT.md

       }
   }
   ```
