```csharp
namespace Bustand.Core;

/// <summary>
/// Marker interface for all Bustand stores.
/// Used for type constraints and discovery.
/// </summary>
public interface IStore
{
    /// <summary>
    /// Event raised when state changes.
    /// </summary>
    event EventHandler? StateChanged;
}

/// <summary>
/// Generic store interface with typed state access.
/// </summary>
/// <typeparam name="TState">The state type, should be a record for immutability.</typeparam>
public interface IStore<out TState> : IStore where TState : class
{
    /// <summary>
    /// Gets the current state. Read-only access.
    /// </summary>
    TState State { get; }
}
```

```csharp
using Microsoft.AspNetCore.Components;

namespace Bustand.Core;

/// <summary>
/// Abstract base class for Bustand stores. Inherit from this class to create a store.
/// </summary>
/// <typeparam name="TState">The state type. Use a C# record for immutability with 'with' expressions.</typeparam>
/// <example>
/// <code>
/// public record CounterState(int Count = 0);
///
/// [BustandStore]
/// public class CounterStore : ZustandStore&lt;CounterState&gt;
/// {
///     public CounterStore() : base(new CounterState()) { }
///
///     public void Increment() => Set(s => s with { Count = s.Count + 1 });
/// }
/// </code>
/// </example>
public abstract class ZustandStore<TState> : IStore<TState> where TState : class
{
    private TState _state;
    private readonly object _lock = new();
    private ComponentBase? _currentComponent;

    /// <inheritdoc />
    public TState State => _state;

    /// <inheritdoc />
    public event EventHandler? StateChanged;

    /// <summary>
    /// Creates a new store with the specified initial state.
    /// </summary>
    /// <param name="initialState">The initial state. Cannot be null.</param>
    /// <exception cref="ArgumentNullException">Thrown when initialState is null.</exception>
    protected ZustandStore(TState initialState)
    {
        _state = initialState ?? throw new ArgumentNullException(nameof(initialState));
    }

    /// <summary>
    /// Updates the state using the provided mutator function.
    /// The mutator receives the current state and should return a new state (use 'with' expression for records).
    /// </summary>
    /// <param name="mutator">A function that takes current state and returns new state.</param>
    /// <example>
    /// <code>
    /// // For record state:
    /// Set(state => state with { Count = state.Count + 1 });
    /// </code>
    /// </example>
    protected void Set(Func<TState, TState> mutator)
    {
        lock (_lock)
        {
            _state = mutator(_state);
        }
        OnStateChanged();
    }

    /// <summary>
    /// Raises the StateChanged event. Uses InvokeAsync when a component context is available
    /// for Blazor Server synchronization context safety (MODE-05 requirement).
    /// </summary>
    /// <remarks>
    /// In Blazor Server, UI updates must occur on the synchronization context.
    /// This method handles that automatically when SetComponentContext has been called.
    /// In WASM or when no component context is set, invokes synchronously.
    /// </remarks>
    protected virtual void OnStateChanged()
    {
        var handler = StateChanged;
        if (handler == null) return;

        // If we have a component context, use InvokeAsync for thread safety
        // This is critical for Blazor Server mode (MODE-05)
        if (_currentComponent != null)
        {
            // Fire and forget - InvokeAsync queues the work on the sync context
            _ = _currentComponent.InvokeAsync(() => handler.Invoke(this, EventArgs.Empty));
        }
        else
        {
            // No component context (e.g., during testing or WASM without context)
            handler.Invoke(this, EventArgs.Empty);
        }
    }

    /// <summary>
    /// Sets the component context for thread-safe state change notifications.
    /// Call this from component's OnInitialized to enable InvokeAsync dispatch.
    /// </summary>
    /// <param name="component">The component subscribing to this store.</param>
    /// <remarks>
    /// This is optional but recommended for Blazor Server mode to ensure
    /// state change notifications are dispatched on the correct synchronization context.
    /// </remarks>
    public void SetComponentContext(ComponentBase component)
    {
        _currentComponent = component;
    }

    /// <summary>
    /// Clears the component context. Call from component's Dispose method.
    /// </summary>
    public void ClearComponentContext()
    {
        _currentComponent = null;
    }
}
```
