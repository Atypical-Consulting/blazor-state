Create a recursive Blazor component that renders JSON as an expandable tree view.

**Component design:**
- Takes `JsonElement` (from System.Text.Json) as parameter
- Recursively renders objects, arrays, and primitives
- Each node tracks its own expand/collapse state
- First level expanded by default (Depth < 1)
- Type-based color coding using CSS classes

**StateTreeView.razor:**
```razor
@using System.Text.Json

@if (Value.ValueKind == JsonValueKind.Object)
{
    <div class="tree-node tree-object">
        <span class="tree-toggle @(IsExpanded ? "expanded" : "")" @onclick="Toggle">
            <span class="tree-bracket">{</span>
        </span>
        @if (IsExpanded)
        {
            <div class="tree-children">
                @foreach (var prop in Value.EnumerateObject())
                {
                    <div class="tree-property">
                        <span class="property-name">"@prop.Name"</span>
                        <span class="property-colon">:</span>
                        <StateTreeView Value="prop.Value" Depth="@(Depth + 1)" />
                    </div>
                }
            </div>
        }
        else
        {
            <span class="tree-collapsed-hint">...</span>
        }
        <span class="tree-bracket">}</span>
    </div>
}
else if (Value.ValueKind == JsonValueKind.Array)
{
    <div class="tree-node tree-array">
        <span class="tree-toggle @(IsExpanded ? "expanded" : "")" @onclick="Toggle">
            <span class="tree-bracket">[</span>
        </span>
        @if (IsExpanded)
        {
            <div class="tree-children">
                @{ var index = 0; }
                @foreach (var item in Value.EnumerateArray())
                {
                    <div class="tree-array-item">
                        <span class="array-index">[@index]</span>
                        <StateTreeView Value="item" Depth="@(Depth + 1)" />
                    </div>
                    index++;
                }
            </div>
        }
        else
        {
            <span class="tree-collapsed-hint">@Value.GetArrayLength() items</span>
        }
        <span class="tree-bracket">]</span>
    </div>
}
else
{
    @* Primitive values *@
    <span class="tree-value @GetValueClass()">@GetDisplayValue()</span>
}

@code {
    [Parameter] public JsonElement Value { get; set; }
    [Parameter] public int Depth { get; set; } = 0;

    private bool IsExpanded { get; set; }

    protected override void OnInitialized()
    {
        // First level expanded by default per CONTEXT.md
        IsExpanded = Depth < 1;
    }

    private void Toggle() => IsExpanded = !IsExpanded;

    private string GetValueClass() => Value.ValueKind switch
    {
        JsonValueKind.String => "value-string",
        JsonValueKind.Number => "value-number",
        JsonValueKind.True or JsonValueKind.False => "value-boolean",
        JsonValueKind.Null => "value-null",
        _ => ""
    };

    private string GetDisplayValue() => Value.ValueKind switch
    {
        JsonValueKind.String => $"\"{Value.GetString()}\"",
        JsonValueKind.Number => Value.GetRawText(),
        JsonValueKind.True => "true",
        JsonValueKind.False => "false",
        JsonValueKind.Null => "null",
        _ => Value.GetRawText()
    };
}
```

Create component with Copy to Clipboard and Download buttons.

**JsonExporter.razor:**
```razor
@inject IJSRuntime JS

<div class="json-exporter">
    <button class="export-btn" @onclick="CopyToClipboard" title="Copy to clipboard">
        <span class="btn-icon">@CopyIcon</span>
        <span class="btn-text">@CopyButtonText</span>
    </button>
    <button class="export-btn" @onclick="DownloadFile" title="Download as JSON">
        <span class="btn-icon">@DownloadIcon</span>
        <span class="btn-text">Download</span>
    </button>
</div>

@code {
    [Parameter] public string Json { get; set; } = "";
    [Parameter] public string FileName { get; set; } = "state.json";

    private string CopyButtonText = "Copy";
    private const string CopyIcon = "&#128203;"; // clipboard icon
    private const string DownloadIcon = "&#128190;"; // floppy disk icon

    private async Task CopyToClipboard()
    {
        try
        {
            await JS.InvokeVoidAsync("navigator.clipboard.writeText", Json);
            CopyButtonText = "Copied!";
            StateHasChanged();

            // Reset after 2 seconds
            await Task.Delay(2000);
            CopyButtonText = "Copy";
            StateHasChanged();
        }
        catch
        {
            CopyButtonText = "Failed";
            StateHasChanged();
        }
    }

    private async Task DownloadFile()
    {
        // Create blob and download via JS
        var dataUrl = $"data:application/json;charset=utf-8,{Uri.EscapeDataString(Json)}";
        await JS.InvokeVoidAsync("eval", $@"
            (function() {{
                var a = document.createElement('a');
                a.href = '{dataUrl}';
                a.download = '{FileName}';
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
            }})();
        ");
    }
}
```

Create StateInspectorPanel that combines tree view and exporter, then wire into DevToolsPage.

**StateInspectorPanel.razor:**
```razor
@using System.Text.Json
@using Bustand.DevTools.Services

@inject IDevToolsStore DevToolsStore

<div class="state-inspector-panel">
    @if (_snapshot != null)
    {
        <JsonExporter Json="@_snapshot.StateJson" FileName="@($"{StoreName}-state.json")" />

        <div class="tree-view-container">
            @if (_parsedJson.HasValue)
            {
                <StateTreeView Value="_parsedJson.Value" />
            }
            else
            {
                <div class="parse-error">
                    <p>Unable to parse state as JSON</p>
                    <pre>@_snapshot.StateJson</pre>
                </div>
            }
        </div>
    }
    else
    {
        <div class="no-state">
            <p>No state recorded yet</p>
        </div>
    }
</div>

@code {
    [Parameter] public string StoreName { get; set; } = "";

    private StateSnapshot? _snapshot;
    private JsonElement? _parsedJson;

    protected override void OnParametersSet()
    {
        LoadSnapshot();
    }

    private void LoadSnapshot()
    {
        _snapshot = DevToolsStore.GetCurrentSnapshot(StoreName);
        if (_snapshot != null)
        {
            try
            {
                using var doc = JsonDocument.Parse(_snapshot.StateJson);
                _parsedJson = doc.RootElement.Clone();
            }
            catch
            {
                _parsedJson = null;
            }
        }
    }
}
```
