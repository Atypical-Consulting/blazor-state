**Updated DevToolsMiddleware:**
```csharp
using Bustand.DevTools.Services;
using Bustand.Middleware;

namespace Bustand.DevTools.Middleware;

/// <summary>
/// Middleware that captures state changes for DevTools.
/// </summary>
/// <typeparam name="TState">The state type.</typeparam>
public class DevToolsMiddleware<TState> : IMiddleware<TState> where TState : class
{
    private readonly IDevToolsStore _devToolsStore;
    private bool _storeRegistered;
    private object? _storeInstance;

    public DevToolsMiddleware(IDevToolsStore devToolsStore)
    {
        _devToolsStore = devToolsStore;
    }

    /// <summary>
    /// Sets the store instance for time-travel support.
    /// Called by DI factory during store construction.
    /// </summary>
    internal void SetStoreInstance(object store)
    {
        _storeInstance = store;
    }

    /// <inheritdoc />
    public bool OnBeforeChange(MiddlewareContext<TState> context)
    {
        // DevTools never blocks state changes
        return true;
    }

    /// <inheritdoc />
    public void OnAfterChange(MiddlewareContext<TState> context)
    {
        // Skip recording during time-travel
        if (_devToolsStore.IsTimeTraveling)
            return;

        // Register store on first state change (lazy registration)
        if (!_storeRegistered && _storeInstance != null)
        {
            var storeName = context.StoreType.Name;
            _devToolsStore.RegisterStore(storeName, _storeInstance);
            _storeRegistered = true;
        }

        // Record the state change
        _devToolsStore.RecordStateChange(
            context.StoreType,
            context.OldState,
            context.NewState,
            context.ActionName,
            context.Timestamp);
    }
}
```

The middleware now:
1. Holds reference to store instance (set by DI)
2. Registers store with DevToolsStore on first state change
3. Continues recording all subsequent changes

```csharp
using Bustand.Configuration;
using Bustand.DevTools.Middleware;
using Bustand.DevTools.Services;
using Microsoft.Extensions.DependencyInjection;
using Microsoft.Extensions.Hosting;

namespace Bustand.DevTools.Extensions;

/// <summary>
/// Extension methods for registering Bustand DevTools.
/// </summary>
public static class DevToolsServiceCollectionExtensions
{
    /// <summary>
    /// Registers Bustand DevTools services.
    /// Call this after AddBustand() and only in development environments.
    /// </summary>
    /// <param name="services">The service collection.</param>
    /// <param name="environment">The host environment.</param>
    /// <returns>The service collection for chaining.</returns>
    /// <example>
    /// <code>
    /// builder.Services.AddBustand(options => options.UseDevTools());
    ///
    /// if (builder.Environment.IsDevelopment())
    /// {
    ///     builder.Services.AddBustandDevTools(builder.Environment);
    /// }
    /// </code>
    /// </example>
    public static IServiceCollection AddBustandDevTools(
        this IServiceCollection services,
        IHostEnvironment environment)
    {
        if (!environment.IsDevelopment())
        {
            Console.WriteLine("[Bustand DevTools] WARNING: DevTools should only be enabled in Development. " +
                "Skipping registration for security.");
            return services;
        }

        return AddDevToolsServices(services);
    }

    /// <summary>
    /// Registers Bustand DevTools services.
    /// Call this after AddBustand() - consumer is responsible for environment check.
    /// </summary>
    public static IServiceCollection AddBustandDevTools(this IServiceCollection services)
    {
        return AddDevToolsServices(services);
    }

    private static IServiceCollection AddDevToolsServices(IServiceCollection services)
    {
        // Register DevToolsStore as Scoped (one per circuit/user)
        services.AddScoped<IDevToolsStore, DevToolsStore>();

        // Register DiffService
        services.AddScoped<DiffService>();

        // Register the open generic DevToolsMiddleware
        // The Bustand core will close it over each store's state type
        services.AddScoped(typeof(DevToolsMiddleware<>));

        return services;
    }
}
```

**Usage pattern:**
```csharp
// Program.cs
builder.Services.AddBustand(options =>
{
    options.UseLogging();
    options.UseDevTools(); // Sets DevToolsEnabled flag
});

if (builder.Environment.IsDevelopment())
{
    builder.Services.AddBustandDevTools(builder.Environment);
}
```

The flow:
1. `UseDevTools()` sets the flag in BustandOptions
2. `AddBustandDevTools()` registers IDevToolsStore and DevToolsMiddleware<>
3. When Bustand creates store pipelines, it sees DevToolsEnabled and adds DevToolsMiddleware to each store's pipeline
