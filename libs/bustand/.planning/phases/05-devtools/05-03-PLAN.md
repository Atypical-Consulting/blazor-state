  <action>
Create StoreSidebar component that shows list of stores with search filter.

**Component structure:**
```razor
@using Bustand.DevTools.Services

<aside class="devtools-sidebar">
    <div class="sidebar-header">
        <h2>Stores</h2>
        <input type="search"
               @bind="SearchFilter"
               @bind:event="oninput"
               placeholder="Filter stores..."
               class="store-search" />
    </div>
    <ul class="store-list">
        @foreach (var storeName in FilteredStores)
        {
            var snapshot = DevToolsStore.GetCurrentSnapshot(storeName);
            <li @key="storeName"
                class="store-item @(SelectedStore == storeName ? "selected" : "")"
                @onclick="() => SelectStore(storeName)">
                <span class="store-name">@storeName</span>
                <span class="store-timestamp">@(snapshot?.Timestamp.ToString("HH:mm:ss"))</span>
            </li>
        }
    </ul>
</aside>

@code {
    [Parameter] public IDevToolsStore DevToolsStore { get; set; } = default!;
    [Parameter] public string? SelectedStore { get; set; }
    [Parameter] public EventCallback<string> OnStoreSelected { get; set; }

    private string SearchFilter = "";

    private IEnumerable<string> FilteredStores =>
        string.IsNullOrWhiteSpace(SearchFilter)
            ? DevToolsStore.RegisteredStoreNames
            : DevToolsStore.RegisteredStoreNames
                .Where(n => n.Contains(SearchFilter, StringComparison.OrdinalIgnoreCase));

    private async Task SelectStore(string storeName)
    {
        await OnStoreSelected.InvokeAsync(storeName);
    }
}
```

Per CONTEXT.md: Fixed width sidebar (280px), search box to filter stores.

**DevToolsPage.razor.cs (code-behind):**
```csharp
public partial class DevToolsPage : IDisposable
{
    private string? SelectedStore;
    private string ActiveTab = "state";

    protected override void OnInitialized()
    {
        DevToolsStore.StateHistoryChanged += OnStateHistoryChanged;
    }

    private void OnStateHistoryChanged(object? sender, EventArgs e)
    {
        InvokeAsync(StateHasChanged);
    }

    private void SelectStore(string storeName)
    {
        SelectedStore = storeName;
    }

    public void Dispose()
    {
        DevToolsStore.StateHistoryChanged -= OnStateHistoryChanged;
    }
}
```
