**DiffService.cs:**
```csharp
using KellermanSoftware.CompareNetObjects;
using System.Text.Json;

namespace Bustand.DevTools.Services;

/// <summary>
/// Service for computing differences between state objects.
/// </summary>
public class DiffService
{
    private readonly CompareLogic _compareLogic;
    private readonly JsonSerializerOptions _jsonOptions;

    public DiffService()
    {
        _compareLogic = new CompareLogic
        {
            Config =
            {
                MaxDifferences = 100,
                IgnoreCollectionOrder = false,
                CompareChildren = true
            }
        };

        _jsonOptions = new JsonSerializerOptions
        {
            WriteIndented = true,
            PropertyNamingPolicy = JsonNamingPolicy.CamelCase
        };
    }

    /// <summary>
    /// Computes the differences between two states.
    /// </summary>
    /// <param name="oldState">The previous state.</param>
    /// <param name="newState">The new state.</param>
    /// <returns>A diff result with categorized changes.</returns>
    public DiffResult ComputeDiff(object? oldState, object? newState)
    {
        if (oldState == null && newState == null)
            return DiffResult.Empty;

        if (oldState == null)
            return new DiffResult
            {
                OldStateJson = "null",
                NewStateJson = SerializeState(newState),
                Differences = new List<DiffItem>
                {
                    new DiffItem("(root)", DiffType.Added, "null", SerializeState(newState))
                }
            };

        if (newState == null)
            return new DiffResult
            {
                OldStateJson = SerializeState(oldState),
                NewStateJson = "null",
                Differences = new List<DiffItem>
                {
                    new DiffItem("(root)", DiffType.Removed, SerializeState(oldState), "null")
                }
            };

        var comparison = _compareLogic.Compare(oldState, newState);

        var differences = comparison.Differences.Select(d => new DiffItem(
            PropertyPath: d.PropertyName,
            Type: DetermineType(d),
            OldValue: d.Object1Value ?? "null",
            NewValue: d.Object2Value ?? "null"
        )).ToList();

        return new DiffResult
        {
            OldStateJson = SerializeState(oldState),
            NewStateJson = SerializeState(newState),
            Differences = differences,
            AreEqual = comparison.AreEqual
        };
    }

    private DiffType DetermineType(Difference d)
    {
        if (d.Object1Value == null || d.Object1Value == "(null)")
            return DiffType.Added;
        if (d.Object2Value == null || d.Object2Value == "(null)")
            return DiffType.Removed;
        return DiffType.Modified;
    }

    private string SerializeState(object? state)
    {
        if (state == null) return "null";
        try
        {
            return JsonSerializer.Serialize(state, _jsonOptions);
        }
        catch
        {
            return state.ToString() ?? "Error serializing";
        }
    }
}

/// <summary>
/// Result of comparing two states.
/// </summary>
public class DiffResult
{
    public string OldStateJson { get; init; } = "";
    public string NewStateJson { get; init; } = "";
    public IReadOnlyList<DiffItem> Differences { get; init; } = Array.Empty<DiffItem>();
    public bool AreEqual { get; init; } = true;

    public static DiffResult Empty => new();
}

/// <summary>
/// A single difference between states.
/// </summary>
public record DiffItem(
    string PropertyPath,
    DiffType Type,
    string OldValue,
    string NewValue
);

/// <summary>
/// Type of difference.
/// </summary>
public enum DiffType
{
    Added,
    Removed,
    Modified
}
```

@using Bustand.DevTools.Services
@using Bustand.DevTools.Models

@inject IDevToolsStore DevToolsStore

<div class="diff-viewer-panel">
    <div class="diff-header">
        <div class="diff-selector">
            <label>Compare:</label>
            <select @bind="LeftIndex">
                @foreach (var snapshot in History)
                {
                    <option value="@snapshot.Index">
                        #@snapshot.Index - @(snapshot.ActionName ?? "Unknown") (@snapshot.Timestamp.ToString("HH:mm:ss"))
                    </option>
                }
            </select>
            <span class="diff-arrow">vs</span>
            <select @bind="RightIndex">
                @foreach (var snapshot in History)
                {
                    <option value="@snapshot.Index">
                        #@snapshot.Index - @(snapshot.ActionName ?? "Unknown") (@snapshot.Timestamp.ToString("HH:mm:ss"))
                    </option>
                }
            </select>
        </div>
    </div>

    @if (DiffResult != null)
    {
        <div class="diff-summary">
            @if (DiffResult.AreEqual)
            {
                <span class="no-changes">No changes</span>
            }
            else
            {
                <span class="change-count">@DiffResult.Differences.Count change(s)</span>
            }
        </div>

        <div class="diff-container">
            <div class="diff-panel diff-left">
                <div class="diff-panel-header">Old State (#@LeftIndex)</div>
                <pre class="diff-content">@DiffResult.OldStateJson</pre>
            </div>
            <div class="diff-panel diff-right">
                <div class="diff-panel-header">New State (#@RightIndex)</div>
                <pre class="diff-content">@DiffResult.NewStateJson</pre>
            </div>
        </div>

        @if (DiffResult.Differences.Count > 0)
        {
            <div class="diff-details">
                <h4>Changes</h4>
                <ul class="diff-list">
                    @foreach (var diff in DiffResult.Differences)
                    {
                        <li class="diff-item @diff.Type.ToString().ToLower()">
                            <span class="diff-path">@diff.PropertyPath</span>
                            <span class="diff-type">@diff.Type</span>
                            @if (diff.Type == DiffType.Modified)
                            {
                                <span class="diff-values">
                                    <span class="old-val">@diff.OldValue</span>
                                    <span class="arrow">-></span>
                                    <span class="new-val">@diff.NewValue</span>
                                </span>
                            }
                            else if (diff.Type == DiffType.Added)
                            {
                                <span class="diff-values">
                                    <span class="new-val">@diff.NewValue</span>
                                </span>
                            }
                            else
                            {
                                <span class="diff-values">
                                    <span class="old-val">@diff.OldValue</span>
                                </span>
                            }
                        </li>
                    }
                </ul>
            </div>
        }
    }
    else
    {
        <div class="no-diff">
            <p>Select two states to compare</p>
        </div>
    }
</div>

@code {
    [Parameter] public string StoreName { get; set; } = "";

    private DiffService _diffService = new();
    private int _leftIndex = 0;
    private int _rightIndex = 0;
    private DiffResult? _diffResult;

    private IReadOnlyList<StateSnapshot> History => DevToolsStore.GetHistory(StoreName);

    private int LeftIndex
    {
        get => _leftIndex;
        set
        {
            _leftIndex = value;
            ComputeDiff();
        }
    }

    private int RightIndex
    {
        get => _rightIndex;
        set
        {
            _rightIndex = value;
            ComputeDiff();
        }
    }

    private DiffResult? DiffResult => _diffResult;

    protected override void OnParametersSet()
    {
        // Default: compare second-to-last with last (most recent change)
        if (History.Count >= 2)
        {
            _leftIndex = History.Count - 2;
            _rightIndex = History.Count - 1;
        }
        else if (History.Count == 1)
        {
            _leftIndex = 0;
            _rightIndex = 0;
        }
        ComputeDiff();
    }

    private void ComputeDiff()
    {
        if (History.Count == 0)
        {
            _diffResult = null;
            return;
        }

        var leftSnapshot = History.ElementAtOrDefault(_leftIndex);
        var rightSnapshot = History.ElementAtOrDefault(_rightIndex);

        if (leftSnapshot == null || rightSnapshot == null)
        {
            _diffResult = null;
            return;
        }

        _diffResult = _diffService.ComputeDiff(leftSnapshot.State, rightSnapshot.State);
    }
}
