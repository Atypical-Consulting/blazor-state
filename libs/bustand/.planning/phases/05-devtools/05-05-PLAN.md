**Complete JumpToState in DevToolsStore:**
```csharp
public void JumpToState(string storeName, int index)
{
    if (!_history.TryGetValue(storeName, out var history))
        return;
    if (index < 0 || index >= history.Count)
        return;
    if (!_stores.TryGetValue(storeName, out var store))
        return;

    var snapshot = history[index];

    _isTimeTraveling = true;
    try
    {
        // Use reflection to call internal SetRestoredState method
        var storeType = store.GetType();
        var setMethod = storeType.GetMethod("SetRestoredState",
            BindingFlags.Instance | BindingFlags.NonPublic);

        if (setMethod != null)
        {
            // We need to set the state, not call SetRestoredState which has early-exit checks
            // Instead, find the _state field and set it directly, then trigger notification
            var stateField = storeType.BaseType?.GetField("_state",
                BindingFlags.Instance | BindingFlags.NonPublic);

            if (stateField != null)
            {
                stateField.SetValue(store, snapshot.State);

                // Trigger OnStateChanged to update subscribers
                var onChangedMethod = storeType.GetMethod("OnStateChanged",
                    BindingFlags.Instance | BindingFlags.NonPublic | BindingFlags.Protected);
                onChangedMethod?.Invoke(store, null);
            }
        }
    }
    finally
    {
        _isTimeTraveling = false;
    }

    _currentIndex[storeName] = index;
    StateHistoryChanged?.Invoke(this, EventArgs.Empty);
}

public void RegisterStore(string storeName, object store)
{
    _stores[storeName] = store;
}

public int GetCurrentIndex(string storeName)
{
    return _currentIndex.TryGetValue(storeName, out var idx) ? idx : -1;
}
```

**Important:** The reflection approach accesses `_state` field in ZustandStore base class. The field is private but accessible via reflection.

**ActionHistoryPanel.razor:**
```razor
@using Bustand.DevTools.Services
@using Bustand.DevTools.Models

@inject IDevToolsStore DevToolsStore

<div class="action-history-panel">
    <div class="history-header">
        <span class="history-count">@History.Count entries</span>
    </div>

    <div class="history-list">
        @if (History.Count == 0)
        {
            <div class="no-history">
                <p>No state changes recorded</p>
            </div>
        }
        else
        {
            @* Display newest first per CONTEXT.md *@
            @foreach (var snapshot in History.Reverse())
            {
                var isCurrent = snapshot.Index == CurrentIndex;
                <div class="history-entry @(isCurrent ? "current" : "")"
                     @onclick="() => JumpTo(snapshot.Index)"
                     @key="snapshot.Index">
                    <div class="entry-header">
                        <span class="entry-action">@(snapshot.ActionName ?? "Unknown")</span>
                        <span class="entry-timestamp">@snapshot.Timestamp.ToString("HH:mm:ss.fff")</span>
                    </div>
                    <div class="entry-index">
                        @if (isCurrent)
                        {
                            <span class="current-marker">current</span>
                        }
                        else
                        {
                            <span class="entry-num">#@snapshot.Index</span>
                        }
                    </div>
                </div>
            }
        }
    </div>
</div>

@code {
    [Parameter] public string StoreName { get; set; } = "";

    private IReadOnlyList<StateSnapshot> History => DevToolsStore.GetHistory(StoreName);
    private int CurrentIndex => DevToolsStore.GetCurrentIndex(StoreName);

    private void JumpTo(int index)
    {
        DevToolsStore.JumpToState(StoreName, index);
    }
}
```

Per CONTEXT.md:
- Newest first (most recent at top) - using `.Reverse()`
- 100-entry limit already enforced by DevToolsStore
