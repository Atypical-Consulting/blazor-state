```csharp
/// <typeparam name="TMiddleware">Middleware type implementing IMiddleware&lt;TState&gt;.</typeparam>
```

**Modify Set(Func<TState, TState> mutator):**
Change from current implementation to:
```csharp
protected void Set(Func<TState, TState> mutator, [CallerMemberName] string? actionName = null)
{
    ThrowIfRendering();
    EnsureStateInitialized();

    TState oldState;
    TState newState;

    lock (_lock)
    {
        oldState = _state!;
        newState = mutator(oldState);
    }

    // Create middleware context
    var context = new MiddlewareContext<TState>
    {
        OldState = oldState,
        NewState = newState,
        StoreType = GetType(),
        ActionName = actionName,
        Timestamp = DateTimeOffset.UtcNow
    };

    // BeforeChange - can block
    if (!_pipeline.InvokeBeforeChange(context))
        return; // State change blocked by middleware

    lock (_lock)
    {
        _state = newState;
    }

    // AfterChange - cannot block
    _pipeline.InvokeAfterChange(context);

    OnStateChanged();
}
```

**Create factory method for building pipelines:**
Add private helper method:
```csharp
private static MiddlewarePipeline<TState> BuildPipeline<TState>(
    IServiceProvider sp,
    IReadOnlyList<Type> middlewareTypes) where TState : class
{
    if (middlewareTypes.Count == 0)
        return MiddlewarePipeline<TState>.Empty;

    var middlewares = new List<IMiddleware<TState>>();
    foreach (var type in middlewareTypes)
    {
        // Middleware types are generic, need to construct concrete type
        var concreteType = type.IsGenericTypeDefinition
            ? type.MakeGenericType(typeof(TState))
            : type;

        if (sp.GetService(concreteType) is IMiddleware<TState> middleware)
        {
            middlewares.Add(middleware);
        }
    }

    return new MiddlewarePipeline<TState>(middlewares);
}
```

```csharp
// After Scrutor scan, wrap store registrations with pipeline injection
if (options.MiddlewareTypes.Count > 0)
{
    var storeTypes = services
        .Where(d => typeof(IStore).IsAssignableFrom(d.ServiceType))
        .Select(d => d.ServiceType)
        .ToList();

    foreach (var storeType in storeTypes)
    {
        var descriptor = services.First(d => d.ServiceType == storeType);
        services.Remove(descriptor);

        services.Add(new ServiceDescriptor(
            storeType,
            sp =>
            {
                var store = ActivatorUtilities.CreateInstance(sp, storeType);
                // Use reflection to call SetPipeline
                InjectPipeline(store, sp, options.MiddlewareTypes);
                return store;
            },
            descriptor.Lifetime));
    }
}
```

Add helper method:
```csharp
private static void InjectPipeline(object store, IServiceProvider sp, List<Type> middlewareTypes)
{
    var storeType = store.GetType();
    var baseType = storeType.BaseType;
    while (baseType != null && !baseType.IsGenericType)
        baseType = baseType.BaseType;

    if (baseType == null) return;

    var stateType = baseType.GetGenericArguments()[0];
    var pipelineType = typeof(MiddlewarePipeline<>).MakeGenericType(stateType);

    // Build middleware list
    var middlewareListType = typeof(List<>).MakeGenericType(
        typeof(IMiddleware<>).MakeGenericType(stateType));
    var middlewares = Activator.CreateInstance(middlewareListType);
    var addMethod = middlewareListType.GetMethod("Add")!;

    foreach (var mwType in middlewareTypes)
    {
        var concreteType = mwType.IsGenericTypeDefinition
            ? mwType.MakeGenericType(stateType)
            : mwType;
        var middleware = sp.GetService(concreteType);
        if (middleware != null)
            addMethod.Invoke(middlewares, new[] { middleware });
    }

    var pipeline = Activator.CreateInstance(pipelineType, middlewares);
    var setPipelineMethod = storeType.GetMethod("SetPipeline",
        System.Reflection.BindingFlags.Instance | System.Reflection.BindingFlags.NonPublic);
    setPipelineMethod?.Invoke(store, new[] { pipeline });
}
```
